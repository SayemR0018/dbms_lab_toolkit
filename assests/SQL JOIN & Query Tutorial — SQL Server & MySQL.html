SQL Server — T-SQL
CREATE TABLE Departments (
    DeptID   INT PRIMARY KEY,
    DeptName NVARCHAR(50),
    Location NVARCHAR(50)
);
CREATE TABLE Employees (
    EmpID    INT PRIMARY KEY,
    EmpName  NVARCHAR(50),
    DeptID   INT,
    Salary   DECIMAL(10,2),
    JoinDate DATE
);
🔵 Uses NVARCHAR for Unicode string support. Standard in SQL Server.
MySQL
CREATE TABLE Departments (
    DeptID   INT PRIMARY KEY,
    DeptName VARCHAR(50),
    Location VARCHAR(50)
) ENGINE=InnoDB;
CREATE TABLE Employees (
    EmpID    INT PRIMARY KEY,
    EmpName  VARCHAR(50),
    DeptID   INT,
    Salary   DECIMAL(10,2),
    JoinDate DATE
) ENGINE=InnoDB;
🟠 MySQL uses VARCHAR (Unicode via utf8mb4 charset). ENGINE=InnoDB enables FK constraints & transactions.

🔍 Basic Query Fundamentals
▶
SELECT Statement
Retrieving columns — basic to advanced projections
DQL
Select all columns — both DBs
-- Wildcard: returns every column
SELECT *
FROM   Employees;
Select specific columns — both DBs
SELECT EmpName, Salary, JoinDate
FROM   Employees
ORDER BY Salary DESC;
Column aliases & expressions — both DBs
SELECT
    EmpName                  AS 'Employee Name',
    Salary,
    Salary * 0.10           AS Bonus,
    Salary + (Salary * 0.10) AS TotalPackage
FROM     Employees
ORDER BY TotalPackage DESC;
SQL Server — TOP N rows
-- Limit result set in SQL Server
SELECT TOP 5
    EmpName, Salary
FROM     Employees
ORDER BY Salary DESC;
🔵 SQL Server uses SELECT TOP N before the column list.
MySQL — LIMIT clause
-- Limit result set in MySQL
SELECT
    EmpName, Salary
FROM     Employees
ORDER BY Salary DESC
LIMIT 5;
🟠 MySQL uses LIMIT N at the end. Add OFFSET n for pagination: LIMIT 5 OFFSET 10.
DISTINCT — remove duplicate values (both DBs)
-- Unique department IDs from the employee table
SELECT DISTINCT DeptID
FROM   Employees
ORDER BY DeptID;
✓ Returns 5 unique DeptIDs (NULL counts as one distinct value)

WHERE Clause
Filter rows using conditions — comparison, logical, pattern
Filtering
Comparison operators — both DBs
-- Salary greater than 60,000
SELECT EmpName, Salary, DeptID
FROM   Employees
WHERE  Salary > 60000
ORDER BY Salary DESC;
✓ Operators: = < > <= >= <> (not equal)
AND / OR / NOT — both DBs
-- Engineering (DeptID=1) AND salary >= 70k
SELECT EmpName, Salary
FROM   Employees
WHERE  DeptID = 1
  AND  Salary >= 70000;

-- HR or Finance employees
SELECT EmpName, DeptID
FROM   Employees
WHERE  DeptID = 2
  OR   DeptID = 3;
LIKE — pattern matching (both DBs)
-- Names starting with 'M'
SELECT EmpName
FROM   Employees
WHERE  EmpName LIKE 'M%';

-- Names containing 'ul'
SELECT EmpName
FROM   Employees
WHERE  EmpName LIKE '%ul%';

-- _ matches exactly one character
SELECT EmpName
FROM   Employees
WHERE  EmpName LIKE '_arida%';
SQL Server — date filter with CAST
-- Employees joined after Jan 2021
SELECT EmpName, JoinDate
FROM   Employees
WHERE  JoinDate > CAST('2021-01-01' AS DATE)
ORDER BY JoinDate;
🔵 SQL Server: use CAST() or CONVERT() for explicit date conversion.
MySQL — date filter
-- Employees joined after Jan 2021
SELECT EmpName, JoinDate
FROM   Employees
WHERE  JoinDate > DATE('2021-01-01')
ORDER BY JoinDate;
🟠 MySQL accepts string literals for DATE columns, but wrapping with DATE() is explicit best practice.

BETWEEN
Inclusive range filter — numbers, dates, text
Range Filter
Numeric range — both DBs
-- Salary between 50k and 70k (inclusive)
SELECT EmpName, Salary
FROM   Employees
WHERE  Salary BETWEEN 50000 AND 70000
ORDER BY Salary;
-- Equivalent to: WHERE Salary >= 50000 AND Salary <= 70000
✓ 7 rows — Karim, Sadia, Shahriar, Tariqul, Nasrin, Laila, Moriam
NOT BETWEEN — exclude a range (both DBs)
-- Employees earning outside mid-range
SELECT EmpName, Salary
FROM   Employees
WHERE  Salary NOT BETWEEN 50000 AND 70000
ORDER BY Salary DESC;
📅
BETWEEN with Dates
Date range queries — SQL Server vs MySQL
Date Range
SQL Server — BETWEEN date range
-- Employees joined in 2020 or 2021
SELECT EmpName, JoinDate
FROM   Employees
WHERE  JoinDate BETWEEN
           CAST('2020-01-01' AS DATE)
       AND CAST('2021-12-31' AS DATE)
ORDER BY JoinDate;
🔵 SQL Server: always cast date strings to avoid implicit conversion surprises.
MySQL — BETWEEN date range
-- Employees joined in 2020 or 2021
SELECT EmpName, JoinDate
FROM   Employees
WHERE  JoinDate BETWEEN
           '2020-01-01'
       AND '2021-12-31'
ORDER BY JoinDate;
🟠 MySQL accepts ISO date strings directly in BETWEEN. Use STR_TO_DATE() for non-ISO formats.
∑
GROUP BY & Aggregate Functions
COUNT, SUM, AVG, MIN, MAX — grouped by category
Aggregation
Basic GROUP BY — both DBs
-- Count employees per department
SELECT
    DeptID,
    COUNT(*) AS EmpCount
FROM     Employees
WHERE    DeptID IS NOT NULL
GROUP BY DeptID
ORDER BY EmpCount DESC;
Multiple aggregates — both DBs
-- Dept-level salary stats
SELECT
    d.DeptName,
    COUNT(e.EmpID) AS HeadCount,
    SUM(e.Salary)  AS TotalSalary,
    AVG(e.Salary)  AS AvgSalary,
    MIN(e.Salary)  AS MinSalary,
    MAX(e.Salary)  AS MaxSalary
FROM       Employees   e
INNER JOIN Departments d ON e.DeptID = d.DeptID
GROUP BY   d.DeptName
ORDER BY   TotalSalary DESC;
SQL Server — ROUND & FORMAT
-- Average salary rounded to 2 decimals
SELECT
    d.DeptName,
    ROUND(AVG(e.Salary), 2) AS AvgSalary
FROM       Employees   e
INNER JOIN Departments d ON e.DeptID = d.DeptID
GROUP BY   d.DeptName
ORDER BY   AvgSalary DESC;
🔵 SQL Server: ROUND(expr, decimals). Use FORMAT(val, 'N2') for display formatting with commas.
MySQL — ROUND & FORMAT
-- Average salary rounded to 2 decimals
SELECT
    d.DeptName,
    ROUND(AVG(e.Salary), 2) AS AvgSalary
FROM       Employees   e
INNER JOIN Departments d ON e.DeptID = d.DeptID
GROUP BY   d.DeptName
ORDER BY   AvgSalary DESC;
🟠 MySQL: ROUND() is identical. Use FORMAT(val, 2) for locale-aware display (returns a string).
⚠ Rule: Every column in SELECT must either be in GROUP BY or wrapped in an aggregate function. Non-aggregated columns outside GROUP BY cause an error (or unpredictable results in MySQL with sql_mode=only_full_group_by disabled).
≥
HAVING Clause
Filter after aggregation — WHERE operates before grouping
Post-Aggregate Filter
Departments with more than 2 employees — both DBs
-- HAVING filters the grouped result
SELECT
    d.DeptName,
    COUNT(e.EmpID)  AS HeadCount,
    SUM(e.Salary)   AS TotalSalary
FROM       Employees   e
INNER JOIN Departments d ON e.DeptID = d.DeptID
GROUP BY   d.DeptName
HAVING     COUNT(e.EmpID) > 2
ORDER BY   HeadCount DESC;
✓ Engineering (4), HR (3), Finance (3)
HAVING with AVG — both DBs
-- Depts where avg salary exceeds 60k
SELECT
    d.DeptName,
    ROUND(AVG(e.Salary), 0) AS AvgSal
FROM       Employees   e
INNER JOIN Departments d ON e.DeptID = d.DeptID
GROUP BY   d.DeptName
HAVING     AVG(e.Salary) > 60000
ORDER BY   AvgSal DESC;
✓ Engineering (74,000), Finance (63,667)
⚡
WHERE vs HAVING
Combining both in one query for precise control
Key Difference
WHERE filters rows first, HAVING filters groups — both DBs
-- WHERE: only include joined-before-2022 employees
-- HAVING: only show depts with >= 2 such employees
SELECT
    d.DeptName,
    COUNT(e.EmpID)  AS EmpCount,
    AVG(e.Salary)   AS AvgSalary
FROM       Employees   e
INNER JOIN Departments d ON e.DeptID = d.DeptID
WHERE      e.JoinDate < '2022-01-01'
GROUP BY   d.DeptName
HAVING     COUNT(e.EmpID) >= 2
ORDER BY   AvgSalary DESC;
⚠ Execution order: FROM → JOIN → WHERE → GROUP BY → HAVING → SELECT → ORDER BY. You cannot use a SELECT alias in HAVING — use the full expression instead.
Quick mental model
-- WHERE  = filter BEFORE grouping (individual rows)
-- HAVING = filter AFTER  grouping (aggregated groups)

SELECT DeptID, COUNT(*) AS Total
FROM   Employees
WHERE  Salary > 40000        -- removes low-earner rows first
GROUP BY DeptID
HAVING COUNT(*) >= 2;        -- then keeps groups with 2+ members

NULL Value Handling
IS NULL, IS NOT NULL, COALESCE, ISNULL / IFNULL, NULLIF
NULL Safety
IS NULL / IS NOT NULL — both DBs
-- Find employees with no department assigned
SELECT EmpName, DeptID
FROM   Employees
WHERE  DeptID IS NULL;

-- Find employees WITH a department
SELECT EmpName, DeptID
FROM   Employees
WHERE  DeptID IS NOT NULL
ORDER BY DeptID;
⚠ Never use = NULL or != NULL — these always return no rows. NULL comparisons require IS NULL / IS NOT NULL.
COALESCE — standard for both DBs
-- Return first non-NULL value from list
SELECT
    EmpName,
    COALESCE(DeptID, 0)       AS SafeDeptID,
    COALESCE(
        CAST(DeptID AS VARCHAR(10)),
        'Unassigned'
    )                          AS DeptLabel
FROM   Employees
ORDER BY DeptID;
✓ COALESCE works identically in SQL Server & MySQL
SQL Server — ISNULL()
-- Replace NULL DeptID with 0
SELECT
    EmpName,
    ISNULL(DeptID, 0)     AS DeptID,
    ISNULL(
        CAST(DeptID AS NVARCHAR(10)),
        N'Unassigned'
    )                      AS DeptLabel
FROM   Employees;

-- NULLIF: return NULL if two values are equal
SELECT
    EmpName,
    NULLIF(DeptID, 0) AS DeptOrNull
FROM   Employees;
🔵 SQL Server ISNULL(col, replacement) — accepts only 2 arguments (use COALESCE for multiple fallbacks).
MySQL — IFNULL() & NULLIF()
-- Replace NULL DeptID with 0
SELECT
    EmpName,
    IFNULL(DeptID, 0)     AS DeptID,
    IFNULL(
        CAST(DeptID AS CHAR),
        'Unassigned'
    )                      AS DeptLabel
FROM   Employees;

-- NULLIF: return NULL if two values are equal
SELECT
    EmpName,
    NULLIF(DeptID, 0) AS DeptOrNull
FROM   Employees;
🟠 MySQL ISNULL(col) only returns 1 or 0 — it does NOT replace the value. Use IFNULL(col, val) instead.
COUNT(*) vs COUNT(col) — NULL behaviour (both DBs)
-- COUNT(*) counts all rows; COUNT(col) ignores NULLs in that column
SELECT
    COUNT(*)       AS TotalRows,      -- 15 (includes Zahir)
    COUNT(DeptID)  AS WithDept,       -- 14 (Zahir excluded — DeptID is NULL)
    COUNT(DISTINCT DeptID) AS UniqueDepts -- 5  (NULL not counted as distinct)
FROM   Employees;
✓ Key rule: aggregate functions (SUM, AVG, MIN, MAX, COUNT(col)) silently skip NULL rows

LEFT JOIN
All left rows, matched or NULL
15 rows 
-- SQL Server: ISNULL()
SELECT
    e.EmpName,
    ISNULL(d.DeptName,
           'Unassigned') AS Department,
    e.Salary
FROM      Employees   e
LEFT JOIN Departments d
       ON e.DeptID = d.DeptID

-- MySQL: IFNULL() or COALESCE()
SELECT
    e.EmpName,

    e.Salary
FROM      Employees   e
LEFT JOIN Departments d
       ON e.DeptID = d.DeptID

FULL OUTER JOIN
Everything from both tables
16 rows 
-- SQL Server: native FULL OUTER JOIN
SELECT
    ISNULL(e.EmpName,  '(No Employee)') AS Employee,
    ISNULL(d.DeptName, '(No Dept)')     AS Department,
    e.Salary,
    d.Location
FROM            Employees   e
FULL OUTER JOIN Departments d
             ON e.DeptID = d.DeptID
ORDER BY        Department;
🔵 SQL Server supports FULL OUTER JOIN natively.
-- MySQL: no FULL OUTER JOIN — use UNION
SELECT
    IFNULL(e.EmpName,  '(No Employee)') AS Employee,
    IFNULL(d.DeptName, '(No Dept)')     AS Department,
    e.Salary, d.Location
FROM      Employees   e
LEFT JOIN Departments d ON e.DeptID = d.DeptID

UNION

SELECT
    IFNULL(e.EmpName,  '(No Employee)'),
    IFNULL(d.DeptName, '(No Dept)'),
    e.Salary, d.Location
FROM       Employees   e
RIGHT JOIN Departments d ON e.DeptID = d.DeptID
ORDER BY   Department;
🟠 MySQL does not support FULL OUTER JOIN. Simulate it using LEFT JOIN UNION RIGHT JOIN.
✓ 16 rows — Zahir + Legal both appear
✕
CROSS JOIN
Full cartesian product — no ON clause
15 × 6 = 90 rows 
-- SQL Server / MySQL — identical syntax
SELECT
    e.EmpName, e.Salary,
    d.DeptName, d.Location
FROM       Employees   e
CROSS JOIN Departments d
ORDER BY   e.EmpName, d.DeptName;

-- Practical: transfer options for Engineering
SELECT
    e.EmpName    AS Employee,
    d.DeptName   AS TransferDept,
    d.Location
FROM       Employees   e
CROSS JOIN Departments d
WHERE      e.DeptID = 1
  AND      d.DeptID <> 1
ORDER BY   e.EmpName;
⚠ No ON clause — always filter with WHERE on large tables. Use for: shift grids, size-colour matrices, schedules.

CASE Statement
Row-level if / else — evaluated per row
Salary Band + Tenure
SQL Server — date comparison with string literal
SELECT
    e.EmpName, d.DeptName, e.Salary,
    CASE
        WHEN e.Salary >= 75000 THEN 'Senior'
        WHEN e.Salary >= 55000 THEN 'Mid-Level'
        WHEN e.Salary >= 40000 THEN 'Junior'
        ELSE                         'Entry'
    END AS SalaryBand,
    CASE
        WHEN e.JoinDate < '2020-01-01' THEN 'Veteran'
        ELSE                                 'Recent'
    END AS Tenure
FROM       Employees e
INNER JOIN Departments d ON e.DeptID = d.DeptID
ORDER BY   e.Salary DESC;
🔵 SQL Server implicitly converts date strings. Best practice: use CAST('2020-01-01' AS DATE) explicitly.
MySQL — use STR_TO_DATE or DATE literal
SELECT
    e.EmpName, d.DeptName, e.Salary,
    CASE
        WHEN e.Salary >= 75000 THEN 'Senior'
        WHEN e.Salary >= 55000 THEN 'Mid-Level'
        WHEN e.Salary >= 40000 THEN 'Junior'
        ELSE                         'Entry'
    END AS SalaryBand,
    CASE
        WHEN e.JoinDate < DATE('2020-01-01') THEN 'Veteran'
        ELSE                                          'Recent'
    END AS Tenure
FROM       Employees e
INNER JOIN Departments d ON e.DeptID = d.DeptID
ORDER BY   e.Salary DESC;
🟠 MySQL accepts '2020-01-01' string literals for DATE columns, but wrapping with DATE() is explicit best practice.

IN & NOT IN
Match against a value list or subquery result
IN — value list (both DBs)
-- Employees in Eng, HR, Finance
SELECT EmpName, DeptID, Salary
FROM     Employees
WHERE    DeptID IN (1, 2, 3)
ORDER BY DeptID, Salary DESC;
NOT IN — value list (both DBs)
-- Exclude Engineering & HR
SELECT EmpName, DeptID, Salary
FROM     Employees
WHERE    DeptID NOT IN (1, 2)
  AND    DeptID IS NOT NULL
ORDER BY Salary DESC;
IN — subquery: Dhaka depts (both DBs)
SELECT EmpName, Salary
FROM   Employees
WHERE  DeptID IN (
    SELECT DeptID
    FROM   Departments
    WHERE  Location = 'Dhaka'
)
ORDER BY Salary DESC;
NOT IN — subquery: empty depts (both DBs)
SELECT DeptName, Location
FROM   Departments
WHERE  DeptID NOT IN (
    SELECT DeptID
    FROM   Employees
    WHERE  DeptID IS NOT NULL
);
-- Result: Legal (no employees)
⚠ NULL trap (both DBs): if the subquery returns even one NULL, NOT IN returns zero rows. Always add WHERE col IS NOT NULL inside the subquery — or use NOT EXISTS.

Faster — no sort
