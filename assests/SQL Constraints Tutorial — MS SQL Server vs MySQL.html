01
PK Primary Key Constraint
Uniquely identifies each row — no duplicates, no NULLs allowed

A Primary Key enforces entity integrity. Each table can have only one Primary Key, which may consist of one or more columns (composite PK). It automatically creates a clustered index in SQL Server and a unique index in MySQL.

MS SQL Server
-- Inline definition
CREATE TABLE Students (
  StudentID   INT           IDENTITY(1,1),
  StudentName NVARCHAR(100) NOT NULL,
  Email       NVARCHAR(150),
  CONSTRAINT PK_Students
    PRIMARY KEY (StudentID)
);

-- Or column-level shorthand
CREATE TABLE Students (
  StudentID   INT PRIMARY KEY
                  IDENTITY(1,1),
  StudentName NVARCHAR(100)
);
MySQL
-- Inline definition
CREATE TABLE Students (
  StudentID   INT           AUTO_INCREMENT,
  StudentName VARCHAR(100)  NOT NULL,
  Email       VARCHAR(150),
  CONSTRAINT PK_Students
    PRIMARY KEY (StudentID)
) ENGINE=InnoDB;

-- Or column-level shorthand
CREATE TABLE Students (
  StudentID   INT PRIMARY KEY
                  AUTO_INCREMENT,
  StudentName VARCHAR(100)
) ENGINE=InnoDB;

02
FK Foreign Key Constraint
Enforces referential integrity between two tables

A Foreign Key links a column (or set of columns) in one table to the Primary Key of another table. It prevents orphan records — you cannot insert a child row that doesn't match a parent, and you cannot delete a parent that has children (unless ON DELETE CASCADE is set).

MS SQL Server
CREATE TABLE Departments (
  DeptID   INT           PRIMARY KEY
                            IDENTITY(1,1),
  DeptName NVARCHAR(100) NOT NULL
);

CREATE TABLE Employees (
  EmpID    INT           PRIMARY KEY
                            IDENTITY(1,1),
  EmpName  NVARCHAR(100) NOT NULL,
  DeptID   INT,
  CONSTRAINT FK_Emp_Dept
    FOREIGN KEY (DeptID)
    REFERENCES Departments(DeptID)
    ON DELETE CASCADE
    ON UPDATE CASCADE
);
MySQL
CREATE TABLE Departments (
  DeptID   INT           PRIMARY KEY
                            AUTO_INCREMENT,
  DeptName VARCHAR(100)  NOT NULL
) ENGINE=InnoDB;

CREATE TABLE Employees (
  EmpID    INT           PRIMARY KEY
                            AUTO_INCREMENT,
  EmpName  VARCHAR(100)  NOT NULL,
  DeptID   INT,
  CONSTRAINT FK_Emp_Dept
    FOREIGN KEY (DeptID)
    REFERENCES Departments(DeptID)
    ON DELETE CASCADE
    ON UPDATE CASCADE
) ENGINE=InnoDB;

03
UK Unique Key Constraint
No duplicate values — but allows a single NULL (differs from PK)

A Unique Key ensures all values in a column (or group of columns) are distinct. Unlike a Primary Key, a Unique column can contain one NULL value in SQL Server, and multiple NULLs in MySQL (since NULLs are not considered equal). A table can have multiple Unique constraints.

MS SQL Server
CREATE TABLE Users (
  UserID    INT           PRIMARY KEY
                              IDENTITY(1,1),
  Username  NVARCHAR(50)  NOT NULL,
  Email     NVARCHAR(150),
  Phone     NVARCHAR(20),
  CONSTRAINT UQ_Users_Email
    UNIQUE (Email),
  CONSTRAINT UQ_Users_Username
    UNIQUE (Username)
);

-- Add unique constraint to existing table
ALTER TABLE Users
  ADD CONSTRAINT UQ_Users_Phone
  UNIQUE (Phone);
MySQL
CREATE TABLE Users (
  UserID    INT           PRIMARY KEY
                              AUTO_INCREMENT,
  Username  VARCHAR(50)   NOT NULL,
  Email     VARCHAR(150),
  Phone     VARCHAR(20),
  CONSTRAINT UQ_Users_Email
    UNIQUE (Email),
  CONSTRAINT UQ_Users_Username
    UNIQUE (Username)
) ENGINE=InnoDB;

-- Add unique constraint to existing table
ALTER TABLE Users
  ADD CONSTRAINT UQ_Users_Phone
  UNIQUE (Phone);

04
CPK Composite Key
A Primary Key (or Unique key) made of two or more columns

A Composite Key uses multiple columns together to uniquely identify a row. Common in junction/bridge tables that represent Many-to-Many relationships (e.g., a student can enroll in many courses, and a course can have many students — the combination of StudentID + CourseID is unique).

MS SQL Server — Enrollment (Junction Table)
CREATE TABLE Enrollment (
  StudentID INT  NOT NULL,
  CourseID  INT  NOT NULL,
  EnrollDate DATE DEFAULT GETDATE(),
  Grade     CHAR(2),

  -- Composite Primary Key
  CONSTRAINT PK_Enrollment
    PRIMARY KEY (StudentID, CourseID),

  CONSTRAINT FK_Enr_Student
    FOREIGN KEY (StudentID)
    REFERENCES Students(StudentID),

  CONSTRAINT FK_Enr_Course
    FOREIGN KEY (CourseID)
    REFERENCES Courses(CourseID)
);
MySQL — Enrollment (Junction Table)
CREATE TABLE Enrollment (
  StudentID INT  NOT NULL,
  CourseID  INT  NOT NULL,
  EnrollDate DATE
    DEFAULT (CURDATE()),
  Grade     CHAR(2),

  -- Composite Primary Key
  CONSTRAINT PK_Enrollment
    PRIMARY KEY (StudentID, CourseID),

  CONSTRAINT FK_Enr_Student
    FOREIGN KEY (StudentID)
    REFERENCES Students(StudentID),

  CONSTRAINT FK_Enr_Course
    FOREIGN KEY (CourseID)
    REFERENCES Courses(CourseID)
) ENGINE=InnoDB;

05
NOT NULL Constraint
Forces a column to always have a value — empty/missing data is not allowed

The NOT NULL constraint prevents a column from holding a NULL value. It's a column-level constraint and can be applied at creation or via ALTER TABLE. Columns without NOT NULL default to NULLable. In SQL Server, use NVARCHAR for Unicode; MySQL uses VARCHAR.

MS SQL Server
CREATE TABLE Products (
  ProductID    INT
                  PRIMARY KEY
                  IDENTITY(1,1),
  ProductName  NVARCHAR(200) NOT NULL,
  SKU          NVARCHAR(50)  NOT NULL,
  Price        DECIMAL(10,2) NOT NULL,
  Description NVARCHAR(MAX) -- NULLable
);

-- Add NOT NULL to existing column
ALTER TABLE Products
ALTER COLUMN SKU NVARCHAR(50) NOT NULL;
MySQL
CREATE TABLE Products (
  ProductID    INT
                  PRIMARY KEY
                  AUTO_INCREMENT,
  ProductName  VARCHAR(200)    NOT NULL,
  SKU          VARCHAR(50)     NOT NULL,
  Price        DECIMAL(10,2)   NOT NULL,
  Description TEXT            -- NULLable
) ENGINE=InnoDB;

-- Add NOT NULL to existing column
ALTER TABLE Products
MODIFY COLUMN SKU VARCHAR(50) NOT NULL;

06
Default Value Constraint
Automatically fills a column with a preset value when no value is provided

The DEFAULT constraint assigns a fallback value to a column when an INSERT statement does not supply a value. Defaults can be literals (e.g., 0, 'Active') or functions (e.g., current timestamp). This is widely used for audit columns like CreatedAt, IsActive, status fields, etc.

MS SQL Server
CREATE TABLE Orders (
  OrderID     INT           PRIMARY KEY
                              IDENTITY(1,1),
  CustomerID  INT           NOT NULL,
  OrderDate   DATETIME
    DEFAULT GETDATE(),
  Status      NVARCHAR(20)
    DEFAULT 'Pending',
  Discount    DECIMAL(5,2)
    DEFAULT 0.00,
  IsActive    BIT
    DEFAULT 1,
  CONSTRAINT FK_Order_Customer
    FOREIGN KEY(CustomerID)
    REFERENCES Customers(CustomerID)
);
MySQL
CREATE TABLE Orders (
  OrderID     INT           PRIMARY KEY
                              AUTO_INCREMENT,
  CustomerID  INT           NOT NULL,
  OrderDate   DATETIME
    DEFAULT CURRENT_TIMESTAMP,
  Status      VARCHAR(20)
    DEFAULT 'Pending',
  Discount    DECIMAL(5,2)
    DEFAULT 0.00,
  IsActive    TINYINT
    DEFAULT 1,
  CONSTRAINT FK_Order_Customer
    FOREIGN KEY(CustomerID)
    REFERENCES Customers(CustomerID)
) ENGINE=InnoDB;

07
CHK Check Constraint
Validates that column values satisfy a specified logical condition

A CHECK constraint lets you define a boolean expression that every row's value must satisfy. If an INSERT or UPDATE would cause the expression to evaluate to FALSE, the operation is rejected. Useful for business rules like age limits, price ranges, status enumerations, and date logic.

MS SQL Server
CREATE TABLE Employees (
  EmpID      INT           PRIMARY KEY
                             IDENTITY(1,1),
  EmpName    NVARCHAR(100) NOT NULL,
  Age        INT,
  Salary     DECIMAL(12,2),
  Gender     CHAR(1),
  JoinDate   DATE,

  CONSTRAINT CHK_Emp_Age
    CHECK (Age BETWEEN 18 AND 65),

  CONSTRAINT CHK_Emp_Salary
    CHECK (Salary >= 5000),

  CONSTRAINT CHK_Emp_Gender
    CHECK (Gender IN ('M', 'F', 'O')),

  CONSTRAINT CHK_Emp_JoinDate
    CHECK (JoinDate >= '2000-01-01')
);
MySQL (8.0.16+ enforced)

-- ① Departments
CREATE TABLE Departments (
  DeptID     INT           PRIMARY KEY IDENTITY(1,1),
  DeptName   NVARCHAR(100) NOT NULL,
  Location   NVARCHAR(100),
  Budget     DECIMAL(15,2),
  EstYear    INT,
  IsActive   BIT           DEFAULT 1,
  CreatedAt  DATETIME      DEFAULT GETDATE(),
  CONSTRAINT CHK_Dept_Budget
    CHECK (Budget > 0)
);

-- ② Faculty
CREATE TABLE Faculty (
  FacultyID   INT           PRIMARY KEY IDENTITY(1,1),
  FacultyName NVARCHAR(100) NOT NULL,
  Email       NVARCHAR(150) NOT NULL,
  DeptID      INT           NOT NULL,
  Designation NVARCHAR(50)  NOT NULL,
  Salary      DECIMAL(12,2),
  JoinDate    DATE          DEFAULT CAST(GETDATE() AS DATE),
  CONSTRAINT UQ_Faculty_Email
    UNIQUE (Email),
  CONSTRAINT FK_Faculty_Dept
    FOREIGN KEY (DeptID)
    REFERENCES Departments(DeptID),
  CONSTRAINT CHK_Faculty_Salary
    CHECK (Salary >= 20000),
  CONSTRAINT CHK_Faculty_Desig
    CHECK (Designation IN (
      'Lecturer', 'Senior Lecturer',
      'Assistant Professor',
      'Associate Professor',
      'Professor'))
);

-- ③ Courses
CREATE TABLE Courses (
  CourseID    INT           PRIMARY KEY IDENTITY(1,1),
  CourseName  NVARCHAR(200) NOT NULL,
  CourseCode  NVARCHAR(20)  NOT NULL,
  DeptID      INT           NOT NULL,
  FacultyID   INT,
  Credits     INT           DEFAULT 3,
  CONSTRAINT UQ_Course_Code
    UNIQUE (CourseCode),
  CONSTRAINT FK_Course_Dept
    FOREIGN KEY (DeptID)
    REFERENCES Departments(DeptID),
  CONSTRAINT FK_Course_Faculty
    FOREIGN KEY (FacultyID)
    REFERENCES Faculty(FacultyID),
  CONSTRAINT CHK_Course_Credits
    CHECK (Credits BETWEEN 1 AND 6)
);

-- ④ Students
CREATE TABLE Students (
  StudentID   INT           PRIMARY KEY IDENTITY(1,1),
  StudentName NVARCHAR(100) NOT NULL,
  Email       NVARCHAR(150) NOT NULL,
  Phone       NVARCHAR(20),
  DeptID      INT           NOT NULL,
  DOB         DATE,
  Gender      CHAR(1),
  EnrollDate  DATE          DEFAULT CAST(GETDATE() AS DATE),
  Status      NVARCHAR(20)  DEFAULT 'Active',
  CONSTRAINT UQ_Student_Email
    UNIQUE (Email),
  CONSTRAINT FK_Student_Dept
    FOREIGN KEY (DeptID)
    REFERENCES Departments(DeptID),
  CONSTRAINT CHK_Student_Gender
    CHECK (Gender IN ('M', 'F', 'O')),
  CONSTRAINT CHK_Student_Status
    CHECK (Status IN (
      'Active', 'Graduated',
      'Suspended', 'Dropped'))

-- ⑤ Enrollment (M:M bridge)
CREATE TABLE Enrollment (
  StudentID  INT          NOT NULL,
  CourseID   INT          NOT NULL,
  Semester   NVARCHAR(20) NOT NULL,
  EnrollDate DATE
    DEFAULT CAST(GETDATE() AS DATE),
  Grade      CHAR(2),
  CONSTRAINT PK_Enrollment
    PRIMARY KEY (StudentID, CourseID),
  CONSTRAINT FK_Enr_Student
    FOREIGN KEY (StudentID)
    REFERENCES Students(StudentID)
    ON DELETE CASCADE,
  CONSTRAINT FK_Enr_Course
    FOREIGN KEY (CourseID)
    REFERENCES Courses(CourseID),
  CONSTRAINT CHK_Enr_Grade
    CHECK (Grade IN (
      'A+','A','A-','B+','B','B-',
      'C+','C','D','F', NULL))
);

-- ⑥ StudentFees (1:1 with Students)
CREATE TABLE StudentFees (
  FeeID       INT           PRIMARY KEY IDENTITY(1,1),
  StudentID   INT           NOT NULL,
  Semester    NVARCHAR(20)  NOT NULL,
  Amount      DECIMAL(10,2) NOT NULL,
  PaidAmount  DECIMAL(10,2) DEFAULT 0.00,
  Status      NVARCHAR(20)  DEFAULT 'Unpaid',
  PaidDate    DATE,
  CONSTRAINT UQ_Fee_Student_Sem
    UNIQUE (StudentID, Semester),
  CONSTRAINT FK_Fee_Student
    FOREIGN KEY (StudentID)
    REFERENCES Students(StudentID),
  CONSTRAINT CHK_Fee_Amount
    CHECK (Amount > 0),
  CONSTRAINT CHK_Fee_Status
    CHECK (Status IN (
      'Paid', 'Partial', 'Unpaid'))
);

-- ⑦ Exams
CREATE TABLE Exams (
  ExamID    INT           PRIMARY KEY IDENTITY(1,1),
  StudentID INT           NOT NULL,
  CourseID  INT           NOT NULL,
  ExamType  NVARCHAR(20)  NOT NULL,
  Marks     DECIMAL(5,2),
  TotalMarks DECIMAL(5,2)  DEFAULT 100,
  ExamDate  DATE
    DEFAULT CAST(GETDATE() AS DATE),
  CONSTRAINT FK_Exam_Student
    FOREIGN KEY (StudentID)
    REFERENCES Students(StudentID),
  CONSTRAINT FK_Exam_Course
    FOREIGN KEY (CourseID)
    REFERENCES Courses(CourseID),
  CONSTRAINT CHK_Exam_Type
    CHECK (ExamType IN (
      'Midterm', 'Final',
      'Quiz', 'Assignment')),
  CONSTRAINT CHK_Exam_Marks
    CHECK (Marks >= 0
      AND Marks <= TotalMarks)
);

  CONSTRAINT CHK_Fee_Status
    CHECK (Status IN (
      'Paid', 'Partial', 'Unpaid'))
) ENGINE=InnoDB;

-- ⑦ Exams
CREATE TABLE Exams (
  ExamID    INT           PRIMARY KEY AUTO_INCREMENT,
  StudentID INT           NOT NULL,
  CourseID  INT           NOT NULL,
  ExamType  VARCHAR(20)   NOT NULL,
  Marks     DECIMAL(5,2),
  TotalMarks DECIMAL(5,2)  DEFAULT 100,
  ExamDate  DATE
    DEFAULT (CURDATE()),
  CONSTRAINT FK_Exam_Student
    FOREIGN KEY (StudentID)
    REFERENCES Students(StudentID),
  CONSTRAINT FK_Exam_Course
    FOREIGN KEY (CourseID)
    REFERENCES Courses(CourseID),
  CONSTRAINT CHK_Exam_Type
    CHECK (ExamType IN (
      'Midterm', 'Final',
      'Quiz', 'Assignment')),
  CONSTRAINT CHK_Exam_Marks
    CHECK (Marks >= 0
      AND Marks <= TotalMarks)
) ENGINE=InnoDB;

Sample Reporting Queries
Student Transcript — MS SQL Server
SELECT
  s.StudentName,
  d.DeptName,
  c.CourseName,
  c.CourseCode,
  e.Semester,
  e.Grade,
  SUM(ex.Marks) AS TotalMarks
FROM Students          s
JOIN Departments        d
  ON s.DeptID    = d.DeptID
JOIN Enrollment         e
  ON s.StudentID = e.StudentID
JOIN Courses             c
  ON e.CourseID  = c.CourseID
LEFT JOIN Exams          ex
  ON ex.StudentID = s.StudentID
  AND ex.CourseID = c.CourseID
WHERE s.StudentID = 1
GROUP BY
  s.StudentName, d.DeptName,
  c.CourseName,  c.CourseCode,
  e.Semester,    e.Grade;

Fee Status Report — MySQL
SELECT
  s.StudentName,
  d.DeptName,
  f.Semester,
  f.Amount       AS TotalFee,
  f.PaidAmount,
  (f.Amount - f.PaidAmount)
                 AS DueAmount,
  f.Status
FROM Students     s
JOIN Departments  d
  ON s.DeptID    = d.DeptID
JOIN StudentFees f
  ON s.StudentID = f.StudentID
WHERE f.Status != 'Paid'
ORDER BY f.Status, s.StudentName;
